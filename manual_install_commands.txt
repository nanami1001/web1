# 手動安裝指令（Windows PowerShell）
# 1) 在專案資料夾執行：
#    cd "C:\Users\user\Downloads\1 (2) (1)\1 (2)\1 (1)"

# 建議（使用虛擬環境，PowerShell）
python -m venv .venv
# 啟用虛擬環境（PowerShell）
.\.venv\Scripts\Activate.ps1
# 升級 pip 與基礎工具
python -m pip install --upgrade pip setuptools wheel
# 安裝 requirements.txt 中列出的所有套件
python -m pip install -r requirements.txt

# 若系統無 python 指令，可使用 py 啟動器替代：
py -3 -m venv .venv
.\.venv\Scripts\Activate.ps1
py -3 -m pip install --upgrade pip setuptools wheel
py -3 -m pip install -r requirements.txt

# 直接系統安裝（不建議；可能需管理員權限）
python -m pip install --upgrade pip setuptools wheel
python -m pip install Flask Flask-SQLAlchemy Flask-Login Flask-Bcrypt Flask-Migrate Flask-WTF WTForms Pillow pyotp qrcode

# 如果遇到 PowerShell 無法執行腳本，可先在當前視窗暫時允許：
# Set-ExecutionPolicy -Scope Process -ExecutionPolicy RemoteSigned

# 完成後，啟動應用範例：
# python run.py
